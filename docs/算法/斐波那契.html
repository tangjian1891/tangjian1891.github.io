排序两数之和斐波那契

斐波那契

形如：[1,1,2,3,5,8,13,21...]给出指定索引，求出索引的值

分析题目：出口为索引0，索引1
//  [1,1,2,3,5,8,13,21]
function foo(index){
  if(index===0||index===1){
    return 1 //出口
  }else{
    return foo(index-1)+foo(index-2) //假设是3，那么就是foo(2)+foo(1) 1+1,依次类推即可
  }
}

console.log(foo(5));// 8
console.log(foo(7));// 21

解法2：使用循环。同样需要记录索引0 和索引1。

function foo(i) {
  let obj = {
    a: 1,
    b: 1,
  };
  if (i === 0 || i === 1) {
    return i;
  }
  let count;
  for (let index = 2; index <= i; index++) {
    count = obj.a + obj.b;
    obj.a = obj.b;
    obj.b = count;
  }
  return count;
}

console.log(foo(5));//8
console.log(foo(7));//21
